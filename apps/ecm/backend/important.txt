app.get('/get-user-with-rights', async (req,res) =>{
  const data = await User.findOne({name:''}).populate({
    path:'staff',
    populate:{
      path:"rights",
      model:'right',
      populate:{
        path:'author',
        model:'User'  
      }
    }
  })
})
